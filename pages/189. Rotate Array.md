- #+BEGIN_QUOTE
  Given an array, rotate the array to the right by k steps, where k is non-negative.
  #+END_QUOTE
- Two solutions
	- Reverse all, then reverse part.
		- ```cpp
		  class Solution {
		  public:
		  void reverse(vector<int>& nums, int start, int end) {
		      int tmp;
		      while (start < end) {
		          tmp = nums[start];
		          nums[start] = nums[end];
		          nums[end] = tmp;
		  
		          start++;
		          end--;
		      }
		  }
		  
		  void rotate(vector<int>& nums, int k) {
		      k %= nums.size();
		      reverse(nums, 0, nums.size() - 1);
		      reverse(nums, 0, k - 1);
		      reverse(nums, k, nums.size() - 1);
		  }
		  };
		  ```
	- **Cycle, GCD** 
	  If we use `+k` cycle to iterate a list of length `n`, we need `GCD(k, n)` cycles start from `[0 : GCD(k, n) - 1]`
		- ```cpp
		  int gcd(int a, int b) {
		      while(true) {
		          if (b == 0) {
		              return a;
		          }
		          a %= b;
		          if (a == 0) {
		              return b;
		          }
		          b %= a;
		      }
		  }
		  
		  void rotate(vector<int>& nums, int k) {
		  
		      int tmpIndex;
		      int tmpNumber;
		      int cycle = gcd(nums.size(), k);
		  
		      for (int i = 0; i < cycle; i++) {
		          tmpIndex = (i + k) % nums.size();
		          while (tmpIndex != i) {
		              tmpNumber = nums[tmpIndex];
		              nums[tmpIndex] = nums[i];
		              nums[i] = tmpNumber;
		              tmpIndex = (tmpIndex + k) % nums.size();
		          }
		      }
		  }
		  ```
